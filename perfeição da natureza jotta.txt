  <script>
    const tabButtons = document.querySelectorAll('.tab-button');
    const tabContents = document.querySelectorAll('.tab-content');
    const progressBar = document.getElementById('progressBar');
    const form = document.getElementById('avaliacaoForm');

    let currentTab = 0;

    function updateProgressBar() {
      const progress = ((currentTab + 1) / tabContents.length) * 100;
      progressBar.style.width = `${progress}%`;
    }

    function openTab(tabId) {
      tabContents.forEach(tab => tab.classList.remove('active'));
      tabButtons.forEach(button => button.classList.remove('active'));
      document.getElementById(tabId).classList.add('active');
      const activeButton = Array.from(tabButtons).find(button => button.textContent.includes(tabId.split('-')[1].charAt(0).toUpperCase() + tabId.split('-')[1].slice(1)));
      if (activeButton) {
        activeButton.classList.add('active');
        currentTab = Array.from(tabButtons).indexOf(activeButton);
        updateProgressBar();
      }
    }

    // Abre a primeira aba por padrão
    openTab('tab-dados');

    // Lógica para exibir/ocultar detalhes da condição médica
    document.querySelectorAll('input[name="condicaoMedica"]').forEach(radio => {
      radio.addEventListener('change', function() {
        document.getElementById('detalheCondicaoDiv').style.display = this.value === "sim" ? "block" : "none";
        if (this.value === "nao") {
          document.getElementById('detalheCondicao').value = "";
        }
      });
    });

    // Lógica para exibir/ocultar detalhes da academia
    document.getElementById('localTreino').addEventListener('change', function() {
      const display = this.value === "academia" || this.value === "ambos" ? "block" : "none";
      document.getElementById('detalheAcademiaDiv').style.display = display;
      if (display === "none") {
        document.getElementById('nomeAcademia').value = "";
        document.getElementById('localizacaoAcademia').value = "";
      }
    });

    // Lógica para exibir feedback da avaliação postural
    document.querySelectorAll('input[name="avaliacaoPostural"]').forEach(radio => {
      radio.addEventListener('change', function() {
        document.getElementById('posturalFeedback').style.display = 'block';
      });
    });

    // Processamento do envio do formulário (mantive a lógica original)
    form.addEventListener('submit', function(event) {
      event.preventDefault();

      let message = "Avaliação de Consultoria Personal\n\n";

      message += "I. Dados Pessoais e Antropométricos:\n";
      message += "Nome: " + document.getElementById('nome').value + "\n";
      message += "Idade: " + document.getElementById('idade').value + "\n";
      message += "Altura: " + document.getElementById('altura').value + " cm\n";
      message += "Peso: " + document.getElementById('peso').value + " kg\n\n";

      let condVal = document.querySelector('input[name="condicaoMedica"]:checked').value;
      message += "II. Histórico e Condição de Saúde:\n";
      message += "Condição médica pré-existente: " + condVal;
      if (condVal === "sim") {
        message += " - " + document.getElementById('detalheCondicao').value;
      }
      message += "\n";
      message += "Lesões anteriores: " + document.getElementById('lesoes').value + "\n";
      message += "Medicação atual: " + document.getElementById('medicacao').value + "\n";
      let fuma = document.querySelector('input[name="fuma"]:checked').value;
      message += "Fuma: " + fuma + "\n";
      message += "Consumo de álcool: " + document.getElementById('alcool').value + "\n";
      message += "Restrição alimentar: " + document.getElementById('restricaoAlimentar').value + "\n";
      message += "Alergias: " + document.getElementById('alergias').value + "\n";
      message += "Histórico familiar: " + document.getElementById('historicoFamiliar').value + "\n\n";

      message += "III. Hábitos de Vida e Qualidade de Vida:\n";
      message += "Qualidade do sono: " + document.getElementById('qualidadeSono').value + "\n";
      message += "Nível de estresse: " + document.getElementById('nivelEstresse').value + "\n";
      message += "Nível de energia: " + document.getElementById('nivelEnergia').value + "\n\n";

      message += "IV. Objetivos e Preferências de Treino:\n";
      message += "Objetivo principal: " + document.getElementById('objetivoPrincipal').value + "\n";
      message += "Frequência do treino: " + document.getElementById('frequenciaTreino').value + " vezes/semana\n";
      message += "Nível de experiência: " + document.getElementById('experiencia').value + "\n";
      message += "Tempo disponível por sessão: " + document.getElementById('tempoSessao').value + "\n";
      let localTreino = document.getElementById('localTreino').value;
      message += "Local de treino: " + localTreino;
      if (localTreino === "academia" || localTreino === "ambos") {
        message += " - Academia: " + document.getElementById('nomeAcademia').value + ", " + document.getElementById('localizacaoAcademia').value;
      }
      message += "\n";
      message += "Preferência de treino: " + document.getElementById('preferenciaTreino').value + "\n";
      message += "Restrição de mobilidade: " + document.getElementById('restricaoMobilidade').value + "\n";
      message += "Áreas problemáticas: " + document.getElementById('areasProblema').value + "\n";
      message += "Tempo sem atividade: " + document.getElementById('tempoInatividade').value + "\n\n";

      message += "Nível de comprometimento: " + document.getElementById('comprometimento').value + "\n";
      message += "Informações adicionais: " + document.getElementById('infoAdicional').value + "\n\n";

      message += "Avaliação Postural: " + postural + "\n";

      responseBox.innerHTML = message + "<br><br><img src='https://i.imgur.com/Kqi7unk.png' alt='OK'>";

      setTimeout(function() {
        let whatsAppNumber = "5585988586381";

        window.open(apiUrl, "_blank");

        responseBox.style.display = "none";
        responseBox.innerHTML = "";
        openTab('tab-dados'); // Retorna para a primeira aba após o envio
      }, 3000);
    });
